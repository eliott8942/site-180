{{- $schedule := .schedule -}} {{- if ne ($schedule | len) 7 -}} {{- errorf "Assertion failed: schedule entries must have 7 entries" -}} {{- end -}} {{- $parsedScheduleSlice := slice -}} {{- range $scheduleDay := $schedule -}} {{- $parsedScheduleDaySlice := slice -}} {{- if $scheduleDay.openingHours -}} {{- range $timeSpan := $scheduleDay.openingHours -}} {{- $start := partial "utils/parse-hour.html" (dict "string" $timeSpan.start) -}} {{- $end := partial "utils/parse-hour.html" (dict "string" $timeSpan.end) -}} {{- if eq (len $parsedScheduleDaySlice) 0 -}} {{- $parsedScheduleDaySlice = slice (slice $start $end) -}} {{- else }} {{- $parsedScheduleDaySlice = $parsedScheduleDaySlice | append (slice $start $end) -}} {{- end -}} {{- end -}} {{- end -}} {{- $sliceToAppend := slice (slice (slice 0 0) (slice 0 0)) | first 0 -}} {{- if ne (len $parsedScheduleDaySlice) 0 -}} {{- $sliceToAppend = $parsedScheduleDaySlice -}} {{- end -}} {{- if eq (len $parsedScheduleSlice) 0 -}} {{- $parsedScheduleSlice = slice ($sliceToAppend) -}} {{- else }} {{- $parsedScheduleSlice = $parsedScheduleSlice | append (($sliceToAppend)) -}} {{- end -}} {{- end -}} {{- return $parsedScheduleSlice -}}
